React Integration

React Hook Pattern

Create a custom hook to manage WebSocket connections in your React app.

hooks/useSocket.ts 
import { useEffect, useState } from 'react';
import { io, Socket } from 'socket.io-client';

export const useSocket = (url: string, username?: string) => {
  const [socket, setSocket] = useState(null);
  const [connected, setConnected] = useState(false);
  const [messages, setMessages] = useState([]);

  useEffect(() => {
    const newSocket = io(url, {
      auth: { username: username || 'Anonymous' }
    });

    newSocket.on('connect', () => {
      setConnected(true);
      console.log('Connected:', newSocket.id);
    });

    newSocket.on('disconnect', () => {
      setConnected(false);
    });

    newSocket.on('message', (data) => {
      setMessages((prev) => [...prev, data]);
    });

    setSocket(newSocket);

    return () => {
      newSocket.close();
    };
  }, [url, username]);

  const sendMessage = (message: string) => {
    if (socket) {
      socket.emit('message', { message });
    }
  };

  return { socket, connected, messages, sendMessage };
};

Server-Sent Events in React

For Redis Pub/Sub, use the EventSource API:

hooks/useSSE.ts 
import { useEffect, useState } from 'react';

export const useSSE = (channel: string) => {
  const [events, setEvents] = useState([]);

  useEffect(() => {
    const eventSource = new EventSource(
      `http://localhost:3000/redis-pubsub/stream?channel=${channel}`
    );

    eventSource.onmessage = (event) => {
      const data = JSON.parse(event.data);
      setEvents((prev) => [...prev, data]);
    };

    eventSource.onerror = () => {
      eventSource.close();
    };

    return () => {
      eventSource.close();
    };
  }, [channel]);

  return events;
};

Vue 3 Integration

composables/useSocket.ts 
import { ref, onMounted, onUnmounted } from 'vue';
import { io, Socket } from 'socket.io-client';

export function useSocket(url: string, username?: string) {
  const socket = ref(null);
  const connected = ref(false);
  const messages = ref([]);

  onMounted(() => {
    socket.value = io(url, {
      auth: { username: username || 'VueUser' }
    });

    socket.value.on('connect', () => {
      connected.value = true;
    });

    socket.value.on('disconnect', () => {
      connected.value = false;
    });

    socket.value.on('message', (data) => {
      messages.value.push(data);
    });
  });

  onUnmounted(() => {
    socket.value?.close();
  });

  const sendMessage = (message: string) => {
    socket.value?.emit('message', { message });
  };

  return {
    socket,
    connected,
    messages,
    sendMessage
  };
}

Angular Integration

services/socket.service.ts 
import { Injectable } from '@angular/core';
import { io, Socket } from 'socket.io-client';
import { BehaviorSubject, Observable } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class SocketService {
  private socket: Socket;
  private connectedSubject = new BehaviorSubject(false);
  private messagesSubject = new BehaviorSubject([]);

  public connected$ = this.connectedSubject.asObservable();
  public messages$ = this.messagesSubject.asObservable();

  constructor() {
    this.socket = io('http://localhost:3000', {
      auth: { username: 'AngularUser' }
    });

    this.socket.on('connect', () => {
      this.connectedSubject.next(true);
    });

    this.socket.on('disconnect', () => {
      this.connectedSubject.next(false);
    });

    this.socket.on('message', (data) => {
      const messages = this.messagesSubject.getValue();
      this.messagesSubject.next([...messages, data]);
    });
  }

  sendMessage(message: string): void {
    this.socket.emit('message', { message });
  }

  disconnect(): void {
    this.socket.close();
  }
}

          type="text"

          type="submit"
          [disabled]="!(socketService.connected$ | async)"
        >
          Send
        </button>
      </form>

export class ChatComponent implements OnDestroy {
  input = '';

  constructor(public socketService: SocketService) {}

  handleSubmit(): void {
    if (this.input.trim()) {
      this.socketService.sendMessage(this.input);
      this.input = '';
    }
  }

  ngOnDestroy(): void {
    this.socketService.disconnect();
  }
}

const socket = io('http://localhost:3000', {
  auth: {
    username: 'VanillaUser'
  }
});

const statusDiv = document.getElementById('status');
const messagesDiv = document.getElementById('messages');
const form = document.getElementById('chat-form');
const input = document.getElementById('message-input');

// Connection status
socket.on('connect', () => {
  statusDiv.textContent = '=â Connected';
  statusDiv.style.color = 'green';
});

socket.on('disconnect', () => {
  statusDiv.textContent = '=4 Disconnected';
  statusDiv.style.color = 'red';
});

// Receive messages
socket.on('message', (data) => {
  const msgDiv = document.createElement('div');
  msgDiv.className = 'message';
  msgDiv.innerHTML = `
    <strong>${data.username}:</strong> ${data.message}
    <span class="time">${new Date(data.timestamp).toLocaleTimeString()}</span>
  `;
  messagesDiv.appendChild(msgDiv);
  messagesDiv.scrollTop = messagesDiv.scrollHeight;
});

// Send messages
form.addEventListener('submit', (e) => {
  e.preventDefault();
  if (input.value.trim()) {
    socket.emit('message', { message: input.value });
    input.value = '';
  }
});

Best Practices

Connection Management

Always clean up connections when components unmount to prevent memory leaks.

Reconnection Logic

Implement automatic reconnection with exponential backoff for better user experience.

Error Handling

Handle connection errors gracefully and show meaningful error messages to users.

Authentication

Send authentication tokens securely and validate on both client and server.

Security Guidelines

  • Always validate and sanitize user input before sending
  • Use HTTPS/WSS in production environments
  • Implement proper authentication and authorization
  • Set CORS policies appropriately
  • Never trust client-side data on the server
  • Implement rate limiting to prevent abuse

Connection Management Tips

Reconnection Strategy

Implement a robust reconnection strategy for better reliability:

Reconnection Example 
const socket = io('http://localhost:3000', {
  reconnection: true,
  reconnectionDelay: 1000,
  reconnectionDelayMax: 5000,
  reconnectionAttempts: 5
});

socket.on('reconnect_attempt', (attemptNumber) => {
  console.log(`Reconnection attempt ${attemptNumber}`);
});

socket.on('reconnect', () => {
  console.log('Reconnected successfully');
});

socket.on('reconnect_failed', () => {
  console.error('Failed to reconnect after maximum attempts');
});

Performance Optimization

  • Batch messages when possible to reduce overhead
  • Use compression for large payloads
  • Implement message throttling for high-frequency updates
  • Clean up event listeners when they're no longer needed
  • Use rooms/channels to target specific users
